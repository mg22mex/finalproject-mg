# Facebook API Credentials
# Copy these values to your .env file or environment variables

# Facebook App Credentials
FACEBOOK_APP_ID=your_facebook_app_id_here
FACEBOOK_APP_SECRET=your_facebook_app_secret_here

# Facebook Access Token (User or Page Access Token)
FACEBOOK_ACCESS_TOKEN=your_facebook_access_token_here

# Facebook Page ID (where posts will be published)
FACEBOOK_PAGE_ID=your_facebook_page_id_here

# Facebook User ID (for Marketplace listings)
FACEBOOK_USER_ID=your_facebook_user_id_here

# Optional: Facebook App Scopes
FACEBOOK_SCOPES=pages_manage_posts,pages_read_engagement,publish_to_groups

# Optional: Facebook API Version
FACEBOOK_API_VERSION=v18.0

# Notes:
# 1. Get these credentials from https://developers.facebook.com/
# 2. Create a Facebook App and get App ID and App Secret
# 3. Generate an Access Token with required permissions
# 4. Find your Page ID in Page Settings > Page Info
# 5. Find your User ID in your profile URL or use Graph API Explorer
